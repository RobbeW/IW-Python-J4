	\def\xscale{1}

		\draw[main, very thin, xstep=1, ystep=1, nearly transparent] (0,0) grid (5,6);

		\foreach \y in {1, ..., 6}{\node[tl,left=1mm] at (0,\y) {$\y$};}

			\node[tl,below=1mm] at (\x, 0) {$\x$};
		}
		\draw[tl] (0, 0) node [below left=1mm] {$0$};
		
		% axes
		\draw[main, ->,thick] (0,0) -- (5,0) node[right] {$x$};
		\draw[main, ->,thick] (0,0) -- (0, 6) node[above] {$y$};
		
		\draw[main, thick] (1,0.9) -- (3,1.1) -- (4,2.5) -- (3.5,5) -- (2,4) -- cycle;
		\fill[BurntOrange] (1,0.9) circle (1.4pt) node [below] {$(x_1,y_1)$};
